#!/bin/bash
#SBATCH --job-name=trimmomatic_RNA_seq_raw
#SBATCH --ntasks=1 #Number of tasks (processes) to be launched
#SBATCH --cpus-per-task=16 #Number of CPUs per task
#SBATCH --mem=64G # Memory allocation
#SBATCH -t 24:00:00
#SBATCH --mail-type=START,END,FAIL
#SBATCH --mail-user=snbogan@ucsb.edu

#Load trimmomatic
module load trimmomatic/0.39

#Set wd
cd /Users/sambogan/Documents/GitHub/Cross_pHox/

mkdir trimmed_fastqs

INPUT_DIR=/Users/sambogan/Documents/GitHub/Cross_pHox/raw_fastqs
OUTPUT_DIR=/Users/sambogan/Documents/GitHub/Cross_pHox/trimmed_fastqs
ADAPTERS=/Users/sambogan/Documents/GitHub/Cross_pHox/Custom_Illumina_Adapter.fa

#Define input files
FILES=($(ls -1 ${INPUT_DIR}/*.fastq.gz))
NUM_FILES=${#FILES[@]}
NUM_JOBS=$(expr ${NUM_FILES} / 2)

#Run Trimmomatic in parallel (2 samples at a time)
parallel --jobs ${NUM_JOBS} \
    'trimmomatic PE -phred33 \
    ${INPUT_DIR}/{} ${INPUT_DIR}/{} \
    ${OUTPUT_DIR}/{}_paired.fastq.gz ${OUTPUT_DIR}/{}_unpaired.fastq.gz \
    ${OUTPUT_DIR}/{}_paired.fastq.gz ${OUTPUT_DIR}/{}_unpaired.fastq.gz \
    ILLUMINACLIP:${ADAPTERS}:2:30:10:8:true \
    LEADING:3 TRAILING:3 SLIDINGWINDOW:4:15 MINLEN:36 HEADCROP:10' ::: ${FILES[@]}

echo "Done."
